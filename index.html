# Clone the GitHub repository to your local machine (replace 'your-
username' and 'avery-austin-website' with your actual username and
repository name)
git clone https://github.com/your-username/avery-austin-website.git
# Navigate to the repository
cd avery-austin-website
# Add your HTML file to the repository
cp /path/to/your/index.html .
# Add the file to the staging area
git add index.html
# Commit your changes
git commit -m "Add index.html"
# Push your code to GitHub
git push -u origin master
